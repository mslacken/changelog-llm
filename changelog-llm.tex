\begin{frame}{Motivation}
  \begin{block}{Changelogs in Open Build Service}
    \begin{itemize}
      \item Package changelogs are created from \texttt{.changes} files
      \item Mostly reflect upstream changelog
      \item \textbf{But} must also contain changes for package creation
    \end{itemize}
  \end{block}
  \begin{block}{The Opportunity}
    \begin{itemize}
      \item Accepted requests to \texttt{openSUSE:Factory} \textbf{are} the right format
      \item Submission requests contain the necessary information
      \item Perfect candidate for LLM automation
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Why use a Local LLM?}
\begin{columns}
\column{0.6\textwidth}
\begin{block}{Existing Tools}
\begin{itemize}
  \item Existing tools can do the same
  \item Cannot capture changes to \texttt{.spec} or \texttt{\_multibuild} files
\end{itemize}
\end{block}
\begin{block}{Advantages}
\begin{itemize}
  \item \textbf{Hype Train}
  \item Pick up semantic changes (e.g., avoid mention Apple-specific changes)
  \item Can be run completely locally, handling embargoed changes
  \item no information leak to cloud models
\end{itemize}
\end{block}
\column{0.3\textwidth}
  \includegraphics[width=\linewidth]{Thought Train}
\end{columns}
\end{frame}


\begin{frame}{Which Model to Choose?}
\begin{columns}
\column{0.8\textwidth}
\begin{block}{Why Fine-Tune Instead of Training from Scratch?}
\begin{itemize}
    \item Pre-trained models understand language structure
    \item Training from scratch requires massive datasets (billions of tokens)
    \item Fine-tuning is computationally cheaper and faster
    \item Specializes existing knowledge for our specific task
\end{itemize}
\end{block}
\begin{block}{Constraints}
\begin{itemize}
    \item Model must be small (around $1\text{ GB}$)
    \item Permissive license
    \item Not quantized, so that it can be fine-tuned
    \item Transformer-based and not a GPT model
\end{itemize}
\end{block}

  \includegraphics[width=\linewidth]{Thinking Chameleon}
\end{columns}
\end{frame}

\begin{frame}{Encoder-Decoder vs. Decoder-Only}
\begin{block}{Encoder-Decoder: T5, BART, mT5}
\begin{itemize}
  \item \textbf{Bidirectional encoder}: Sees entire input at once
  \item Best for: Translation, summarization, structured text generation
\end{itemize}
\end{block}

\begin{block}{Decoder-Only: GPT-3, GPT-4, LLaMA, Mistral}
\begin{itemize}
  \item \textbf{Single decoder}: Input and output in same stream, but only sees previous tokens
  \item Best for: Text completion, general-purpose chat
\end{itemize}
\end{block}

\begin{alertblock}{Key Difference}
T5 can \textbf{read all input first}, then generate. GPT processes input and output sequentially. This makes T5 more efficient for tasks with structured input.
\end{alertblock}
\end{frame}

\begin{frame}{T5 Architecture: Encoder-Decoder}
\begin{columns}[t]
\column{0.5\textwidth}
\begin{block}{Encoder-Decoder Design}
\begin{itemize}
  \item \textbf{Encoder}: Processes input text
  \begin{itemize}
    \item Bidirectional attention
    \item Understands full context
    \item Creates rich representations
  \end{itemize}
  \item \textbf{Decoder}: Generates output text
  \begin{itemize}
    \item Autoregressive generation
    \item Attends to encoder outputs
    \item Produces token-by-token
  \end{itemize}
\end{itemize}
\end{block}

\begin{frame}[fragile]{OBS Data Extraction}
\begin{block}{What data do we need?}
\begin{itemize}
  \item Accepted requests to \texttt{openSUSE:Factory}
  \item Spec file diffs (version updates, build changes)
  \item Source file changes (added, removed, modified)
  \item Release notes from upstream (GitHub, archive changelogs)
  \item The actual \texttt{.changes} file entries (our target)
\end{itemize}
\end{block}

\begin{frame}{OBS Data Extraction}
\begin{columns}
\column{0.6\textwidth}
\begin{block}{Naive Approach}
  \begin{itemize}
    \item Choose a random number smaller than the latest request
    \item Check if submission went to \texttt{openSUSE:Factory} and was accepted
    \item Identify \texttt{Source0}
    \item Extract first 20 lines of \texttt{CHANGELOG} \ldots

\begin{frame}{What is Tokenization?}
\begin{columns}
\column{0.5\textwidth}
\begin{block}{Tokenization}
Converts text into numerical representations that models can process.
\end{block}

\begin{block}{How It Works}
\begin{itemize}
  \item Text split into subword units (tokens)
  \item Each token mapped to unique ID
  \item Vocabulary built from training corpus
  \item Unknown words broken into known subwords
\end{itemize}
\end{block}

\column{0.5\textwidth}
\centering
\begin{tikzpicture}[scale=0.9, every node/.style={font=\sffamily\small}]
  % Input text
  \node[draw, rounded corners, fill=blue!10, text width=5cm, align=center] (text) at (0,0)
    {\texttt{"Update to version 45.0"}};

  % Arrow down
  \draw[->, thick] (text.south) -- ++(0,-0.5) node[midway, right] {\tiny Tokenize};

  % Tokens
  \node[draw, rounded corners, fill=green!10, text width=5cm, align=center] (tokens) at (0,-2)
    {\texttt{["Update", "to", "version", "45", ".", "0"]}};

  % Arrow down
  \draw[->, thick] (tokens.south) -- ++(0,-0.5) node[midway, right] {\tiny Map to IDs};

  % Token IDs
  \node[draw, rounded corners, fill=orange!10, text width=5cm, align=center] (ids) at (0,-4)
    {\texttt{[5974, 12, 988, 1902, 5, 209]}};

  % Example of subword tokenization
  \node[draw, rounded corners, fill=yellow!10, text width=5cm, align=left, font=\sffamily\tiny] (subword) at (0,-5.5)
    {Subword example:\\
    \texttt{"changelog" $\rightarrow$ ["change", "log"]}\\
    \texttt{"updating" $\rightarrow$ ["updat", "ing"]}};
\end{tikzpicture}
\end{columns}
\end{frame}

\begin{frame}{Data Preprocessing \& Tokenization}
\begin{block}{Special Token Handling}
\begin{itemize}
  \item Newline character (\texttt{\textbackslash n}) added as special token
  \item Preserves structure in multi-line changelogs
\end{itemize}
\end{block}

\begin{block}{Sequence Length Limits}
\begin{itemize}
  \item \textbf{Input}: Maximum 1024 tokens (spec diffs + changelog)
  \item \textbf{Target}: Maximum 512 tokens (generated changelog entry)
  \item Sequences exceeding limits are truncated
\end{itemize}
\end{block}

\begin{block}{Padding Strategy}
\begin{itemize}
  \item Dynamic padding: sequences padded to \texttt{longest} in batch
  \item Padding tokens in labels replaced with $-100$
  \item Loss function ignores $-100$ (no gradient from padding)
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{What is Quantization?}
\begin{block}{Quantization}
Reduces numerical precision of weights from high precision (Float32) to lower (Int8).
\end{block}

\begin{columns}[t]
\column{0.5\textwidth}
\begin{block}{Why It Works}
\begin{itemize}
  \item Neural networks are \textbf{over-parametrized}
  \item Robust features tolerate small numerical errors
  \item Weight values cluster
  \item Activations follow predictable distributions
\end{itemize}
\end{block}

\column{0.5\textwidth}
\begin{block}{Benefits}
\begin{itemize}
  \item \textbf{Smaller size}: Float32 $\rightarrow$ Int8 = 4× reduction
  \item \textbf{Faster inference}: Integer operations are faster
  \item \textbf{Lower memory}: Fits on consumer hardware
  \item \textbf{Minimal accuracy loss}: Usually <2\% degradation
\end{itemize}
\end{block}
\end{columns}

\end{frame}

\begin{frame}{Quantization Implementation}
\begin{columns}
\column{0.6\textwidth}
\begin{block}{CTranslate2 vs. Ollama}
\begin{itemize}
  \item \textbf{Ollama}: Optimized for decoder-only models (GPT, LLaMA)
  \item \textbf{CTranslate2}: Optimized for encoder-decoder models (T5, BART)
  \item T5's bidirectional encoder requires specialized attention mechanisms
\end{itemize}
\end{block}

\begin{block}{INT8 Quantization}
\begin{itemize}
  \item Float32 (4 bytes) $\rightarrow$ Int8 (1 byte)
  \item \textbf{Result}: $\sim$4× smaller model size
  \item \textbf{t5-base}: 892 MB $\rightarrow$ $\sim$220 MB
  \item \textbf{t5-large}: 2.75 GB $\rightarrow$ $\sim$690 MB
\end{itemize}
\end{block}
