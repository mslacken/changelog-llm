

\begin{frame}{T5 Architecture: Encoder-Decoder}
\begin{columns}
\column{0.5\textwidth}
\begin{block}{Encoder-Decoder Design}
\begin{itemize}
  \item \textbf{Encoder}: Processes input text
  \begin{itemize}
    \item Bidirectional attention
    \item Understands full context
    \item Creates rich representations
  \end{itemize}
  \item \textbf{Decoder}: Generates output text
  \begin{itemize}
    \item Autoregressive generation
    \item Attends to encoder outputs
    \item Produces token-by-token
  \end{itemize}
\end{itemize}
\end{block}
\column{0.5\textwidth}
\begin{block}{Architecture Components}
\begin{itemize}
  \item Self-attention layers
  \item Cross-attention (decoder $\rightarrow$ encoder)
  \item Feed-forward networks
  \item Layer normalization
  \item Relative position embeddings
\end{itemize}
\end{block}

\begin{block}{Model Sizes}
\begin{itemize}
  \item Small: 12 layers (enc+dec)
  \item Base: 24 layers
  \item Large: 48 layers
  \item XL: 96 layers (3B params)
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{block}{Encoder Advantage}
\begin{itemize}
  \item Reads \textbf{all} input simultaneously
  \item Understands relationships between parts
  \item Efficient cross-attention
  \item No wasted context on input
\end{itemize}
\end{block}
\column{0.5\textwidth}
\begin{block}{GPT Limitations}
With decoder-only:
\begin{itemize}
  \item Input consumes output budget
  \item 1024 input + 512 output = 1536 total
  \item Sequential processing is slower
  \item Less efficient for structured inputs
\end{itemize}
\end{block}

\begin{block}{T5 Efficiency}
\begin{itemize}
  \item 1024 input (encoder)
  \item + 512 output (decoder)
  \item Separate budgets!
  \item Parallel encoding
\end{itemize}
\end{block}

    cmd = [req_getter, str(req_id), "-f", "state=accepted",
           "-f", "target=openSUSE:Factory"]
    data = json.loads(run_command(cmd))

    # 2. Extract source details
    action = data["Actions"][0]
    project = action["Source"]["Project"]
    package = action["Source"]["Package"]
    revision = action["Source"]["Rev"]

    # 3. Extract diffs from sourcediff

        "spec_diff": spec_diff,
        "changes_diff": changes_diff,  # Our target!
        "archive_changelog": ext_data["archive_changelog"],

\begin{frame}[fragile]{Target Processing: Changes Diff}
\begin{lstlisting}[language=Python]
def preprocess_target_function(data):
    """
    Extract the actual changelog entry from .changes diff.
    Removes the leading '+' sign from diff lines.
    Preserves newlines explicitly.
    """
    results = []
    for diff in data['changes_diff']:
        processed_lines = []
        for line in diff.split("\n"):
            if line.startswith("+"):
                content = line[1:]  # Remove '+'
                if content.strip():
                    processed_lines.append(content)

        # Join with newline character
        results.append("\n".join(processed_lines))

    return results
\end{lstlisting}

\begin{block}{Key Insight}
Newlines are semantic in changelog entries! Must be preserved.

\section{Special Considerations}
\begin{frame}[fragile]{The Newline Token Challenge}
\begin{block}{Problem}
Changelog entries have semantic structure encoded in newlines:

\texttt{- update to version 2.0:}

\texttt{~~* Fix critical bug}

\texttt{~~* Add new feature}

But T5 tokenizer doesn't preserve \texttt{\textbackslash n} well by default!
\end{block}

\begin{block}{Solution}
\begin{lstlisting}[language=Python]
tokenizer = AutoTokenizer.from_pretrained(model_id)
tokenizer.add_special_tokens({'additional_special_tokens': ['\n']})

# Resize model embeddings
model = T5ForConditionalGeneration.from_pretrained(model_id)
model.resize_token_embeddings(len(tokenizer))
\end{lstlisting}
\end{block}

\begin{alertblock}{Critical}
Must decode outputs carefully to preserve the \texttt{\textbackslash n} token!
\end{alertblock}
\end{frame}

\begin{frame}[fragile]{Decoding with Newline Preservation}
\begin{lstlisting}[language=Python, basicstyle=\ttfamily\small]
def decode_changelog(output_ids):
    import re

    # Decode without removing special tokens first
    decoded = tokenizer.decode(output_ids,
                              skip_special_tokens=False)

    # Remove only standard T5 tokens, keep our custom \n token
    decoded = decoded.replace(tokenizer.pad_token, "")
    decoded = decoded.replace(tokenizer.eos_token, "")

    # Remove sentinel tokens (e.g., <extra_id_0>)
    decoded = re.sub(r"<extra_id_\d+>", "", decoded)

    return decoded.strip()
\end{lstlisting}

\begin{block}{Result}
Properly formatted, multi-line changelog entries!
\end{block}
\end{frame}

\section{Model Optimization}
\begin{frame}{Quantization: Making Models Smaller}
\begin{block}{What is Quantization?}
Reduces numerical precision of model weights:
\begin{itemize}
  \item \textbf{float32}: Full precision (baseline)
  \item \textbf{float16}: Half precision (GPU only, 50\% size)
  \item \textbf{int8}: 8-bit integers (75\% size reduction)
\end{itemize}
\end{block}

\begin{block}{Benefits}
\begin{itemize}
  \item \textbf{Smaller models}: 75\% size reduction with int8
  \item \textbf{Faster inference}: Fewer memory operations
  \item \textbf{Lower memory}: Fits on smaller GPUs or CPU
  \item \textbf{Minimal quality loss}: Usually <1\% accuracy drop
\end{itemize}
\end{block}

\begin{block}{Trade-offs}
\begin{itemize}
  \item Some quantization requires specific hardware (GPU for float16)
  \item int8 works on both CPU and GPU
\end{itemize}
\end{block}
\end{frame}
